            <textarea id="newsContent" rows="4" placeholder="Escribe la noticia aquí..." required></textarea>
            <button type="submit">Publicar noticia</button>
        </form>

        <!-- Input para ingresar la contraseña -->
        <div class="password-input">
            <input type="password" id="password" placeholder="Ingresa la contraseña para añadir una noticia" />
            <button onclick="checkPassword()">Ingresar</button>
        </div>

        <!-- Aquí aparecerán las noticias -->

            <!-- Las noticias se agregarán aquí -->
        </div>

    <script>
        // Array temporal para almacenar las noticias
        let noticias = [];

        // Selección de los elementos
        const newsForm = document.getElementById('newsForm');
        const newsItems = document.getElementById('newsItems');
        const passwordInput = document.querySelector('.password-input');

        // Función para verificar la contraseña
        function checkPassword() {
            const password = document.getElementById('password').value;
            if (password === 'diplomado') {
                // Mostrar el formulario de noticias si la contraseña es correcta
                newsForm.style.display = 'block';
                passwordInput.style.display = 'none';
            } else {
                alert('Contraseña incorrecta');
            }
        }

        // Manejador del evento submit del formulario de noticias
        newsForm.addEventListener('submit', function (e) {
            e.preventDefault();

            // Obtener los valores del formulario
            const title = document.getElementById('newsTitle').value;
            const content = document.getElementById('newsContent').value;

            // Crear una nueva noticia en formato de objeto
            const nuevaNoticia = {
                titulo: title,
                contenido: content
            };

            // Agregar la noticia al array de noticias
            noticias.push(nuevaNoticia);

            // Renderizar las noticias
            renderizarNoticias();

            // Limpiar el formulario
            newsForm.reset();
        });

        // Función para renderizar las noticias en el DOM
        function renderizarNoticias() {
            newsItems.innerHTML = '';

            noticias.forEach(noticia => {
                const item = document.createElement('div');
                item.classList.add('news-item');
                item.innerHTML = `
                    <h3>${noticia.titulo}</h3>
                    <p>${noticia.contenido}</p>
                `;

                // Agregar la noticia al contenedor
                newsItems.appendChild(item);
            });
        }
    </script>
