        <h1>Foro de Aportes</h1>

        <!-- Formulario para agregar nuevos mensajes -->
        <form class="forum-form" id="forumForm">
            <input type="text" id="username" placeholder="Tu nombre (puedes usar cualquier nombre)" required>
            <textarea id="message" rows="4" placeholder="Escribe tu mensaje aquí..." required></textarea>
            <button type="submit">Enviar mensaje</button>
        </form>

        <!-- Aquí aparecerán los mensajes -->
        <div id="forumPosts">
            <!-- Los mensajes se agregarán aquí -->
        </div>
    </div>

    <script>
        // Selección de los elementos
        const forumForm = document.getElementById('forumForm');
        const forumPosts = document.getElementById('forumPosts');

        // Almacenar los mensajes en un array temporal para simular una base de datos
        let mensajes = [];

        // Manejador del evento submit del formulario
        forumForm.addEventListener('submit', function (e) {
            e.preventDefault();

            // Obtener el valor del usuario y del mensaje
            const username = document.getElementById('username').value;
            const message = document.getElementById('message').value;

            // Crear el nuevo post en formato de objeto y agregarlo a la lista de mensajes
            const nuevoMensaje = {
                usuario: username,
                contenido: message,
                votos: 0
            };

            mensajes.push(nuevoMensaje);

            // Renderizar el foro nuevamente para mostrar el nuevo mensaje
            renderizarForo();

            // Limpiar el formulario
            forumForm.reset();
        });

        // Función para renderizar todos los mensajes del foro
        function renderizarForo() {
            forumPosts.innerHTML = '';

            mensajes.forEach((mensaje, index) => {
                const post = document.createElement('div');
                post.classList.add('forum-post');
                post.innerHTML = `
                    <h3>${mensaje.usuario}</h3>
                    <p>${mensaje.contenido}</p>
                    <div class="actions">
                        <button class="vote-btn" onclick="upvote(${index})">👍 Votar</button>
                        <span class="vote-count">${mensaje.votos}</span>
                    </div>
                `;

                // Agregar el post al contenedor de mensajes
                forumPosts.appendChild(post);
            });
        }

        // Función para incrementar el voto
        function upvote(index) {
            mensajes[index].votos++;
            renderizarForo();
        }
    </script>
